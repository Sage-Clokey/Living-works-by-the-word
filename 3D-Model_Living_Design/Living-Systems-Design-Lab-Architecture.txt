import math
import random
from dataclasses import dataclass, field
from typing import List, Optional, Tuple

import numpy as np
import pyvista as pv


# ============================================================
# Utility functions
# ============================================================

def normalize(v: np.ndarray) -> np.ndarray:
    norm = np.linalg.norm(v)
    if norm < 1e-12:
        return v.copy()
    return v / norm


def random_unit_vector() -> np.ndarray:
    v = np.random.normal(size=3)
    return normalize(v)


def rotate_vector(vec: np.ndarray, axis: np.ndarray, angle_rad: float) -> np.ndarray:
    """
    Rodrigues' rotation formula.
    """
    axis = normalize(axis)
    vec = np.asarray(vec, dtype=float)
    cos_a = math.cos(angle_rad)
    sin_a = math.sin(angle_rad)
    return (
        vec * cos_a
        + np.cross(axis, vec) * sin_a
        + axis * np.dot(axis, vec) * (1.0 - cos_a)
    )


def orthogonal_vector(v: np.ndarray) -> np.ndarray:
    """
    Returns some vector orthogonal to v.
    """
    v = normalize(v)
    if abs(v[0]) < 0.9:
        other = np.array([1.0, 0.0, 0.0])
    else:
        other = np.array([0.0, 1.0, 0.0])
    ortho = np.cross(v, other)
    return normalize(ortho)


def distance_point_to_segment(p: np.ndarray, a: np.ndarray, b: np.ndarray) -> float:
    """
    Compute distance from point p to line segment ab.
    """
    ab = b - a
    denom = np.dot(ab, ab)
    if denom < 1e-12:
        return np.linalg.norm(p - a)
    t = np.dot(p - a, ab) / denom
    t = max(0.0, min(1.0, t))
    closest = a + t * ab
    return np.linalg.norm(p - closest)

# Data structures
# ============================================================

@dataclass
class Segment:
    start: np.ndarray
    end: np.ndarray
    radius: float
    parent_id: Optional[int] = None


@dataclass
class Tip:
    position: np.ndarray
    direction: np.ndarray
    radius: float
    age: int = 0
    alive: bool = True
    lineage_depth: int = 0


@dataclass
class GrowthParams:
    mode: str = "tree"  # tree, coral, spiral

    steps: int = 300
    step_size: float = 0.25

    branching_probability: float = 0.08
    max_tips: int = 128
    max_depth: int = 8

    upward_bias: float = 0.7          # tendency toward +Z
    radial_bias: float = 0.0          # tendency away from origin
    twist_strength: float = 0.0       # tendency to rotate around Z
    noise_strength: float = 0.2       # random wiggle

    branch_angle_deg: float = 28.0
    split_probability: float = 0.5    # probability to make two child branches

    taper: float = 0.985              # how quickly branches thin
    min_radius: float = 0.015
    initial_radius: float = 0.08

    collision_avoidance: bool = True
    collision_distance: float = 0.12

    kill_if_outside_radius: Optional[float] = None

    tropism_target: Optional[np.ndarray] = None  # optional attractor

    seed: int = 42


# ============================================================
# Engine
# ============================================================

class MorphogenesisEngine:
    def __init__(self, params: GrowthParams):
        self.params = params
        self.rng = random.Random(params.seed)
        np.random.seed(params.seed)

        self.segments: List[Segment] = []
        self.tips: List[Tip] = []

        self.origin = np.array([0.0, 0.0, 0.0], dtype=float)

        self._init_structure()

    def _init_structure(self) -> None:
        initial_dir = np.array([0.0, 0.0, 1.0], dtype=float)

        if self.params.mode == "coral":
            initial_dir = normalize(np.array([0.15, 0.05, 1.0]))
        elif self.params.mode == "spiral":
            initial_dir = normalize(np.array([0.5, 0.0, 1.0]))

        self.tips.append(
            Tip(
                position=self.origin.copy(),
                direction=initial_dir,
                radius=self.params.initial_radius,
                age=0,
                alive=True,
                lineage_depth=0,
            )
        )

    def _mode_defaults_direction_adjustment(self, pos: np.ndarray, direction: np.ndarray) -> np.ndarray:
        """
        Adds different behavior based on selected growth mode.
        """
        d = direction.copy()

        if self.params.mode == "tree":
            # Strong upward bias, slight radial spread
            d += np.array([0.0, 0.0, 1.0]) * self.params.upward_bias
            radial = pos.copy()
            radial[2] = 0.0
            if np.linalg.norm(radial) > 1e-9:
                d += normalize(radial) * self.params.radial_bias

        elif self.params.mode == "coral":
            # Upward but more noisy and outward
            d += np.array([0.0, 0.0, 1.0]) * (0.6 * self.params.upward_bias)
            radial = pos - self.origin
            if np.linalg.norm(radial) > 1e-9:
                d += normalize(radial) * max(self.params.radial_bias, 0.25)

        elif self.params.mode == "spiral":
            # Grow upward while twisting around z-axis
            d += np.array([0.0, 0.0, 1.0]) * self.params.upward_bias

            xy = pos.copy()
            xy[2] = 0.0
            if np.linalg.norm(xy) > 1e-9:
                tangent = np.array([-xy[1], xy[0], 0.0], dtype=float)
                tangent = normalize(tangent)
                d += tangent * max(self.params.twist_strength, 0.4)
                d += normalize(xy) * max(self.params.radial_bias, 0.15)
            else:
                d += np.array([1.0, 0.0, 0.0]) * 0.15

        return normalize(d)

    def _apply_tropism(self, pos: np.ndarray, direction: np.ndarray) -> np.ndarray:
        d = direction.copy()
        if self.params.tropism_target is not None:
            toward_target = self.params.tropism_target - pos
            if np.linalg.norm(toward_target) > 1e-12:
                d += normalize(toward_target) * 0.25
        return normalize(d)

    def _apply_noise(self, direction: np.ndarray) -> np.ndarray:
        noise = random_unit_vector() * self.params.noise_strength
        return normalize(direction + noise)

    def _apply_collision_avoidance(self, pos: np.ndarray, proposed_dir: np.ndarray) -> np.ndarray:
        """
        Push away from nearby segments.
        """
        if not self.params.collision_avoidance or not self.segments:
            return proposed_dir

        repel = np.zeros(3, dtype=float)
        sample_point = pos + proposed_dir * self.params.step_size

        for seg in self.segments[-500:]:
            dist = distance_point_to_segment(sample_point, seg.start, seg.end)
            if dist < self.params.collision_distance:
                midpoint = 0.5 * (seg.start + seg.end)
                away = sample_point - midpoint
                if np.linalg.norm(away) > 1e-12:
                    strength = (self.params.collision_distance - dist) / self.params.collision_distance
                    repel += normalize(away) * strength

        if np.linalg.norm(repel) > 1e-12:
            proposed_dir = normalize(proposed_dir + 0.8 * repel)

        return proposed_dir

    def _should_kill_tip(self, new_pos: np.ndarray, new_radius: float, depth: int) -> bool:
        if new_radius < self.params.min_radius:
            return True

        if depth > self.params.max_depth:
            return True

        limit = self.params.kill_if_outside_radius
        if limit is not None and np.linalg.norm(new_pos - self.origin) > limit:
            return True

        return False

    def _grow_tip_once(self, tip: Tip, tip_index: int) -> List[Tip]:
        if not tip.alive:
            return []

        d = tip.direction.copy()
        d = self._mode_defaults_direction_adjustment(tip.position, d)
        d = self._apply_tropism(tip.position, d)
        d = self._apply_noise(d)
        d = self._apply_collision_avoidance(tip.position, d)

        start = tip.position.copy()
        end = start + d * self.params.step_size
        new_radius = tip.radius * self.params.taper

        if self._should_kill_tip(end, new_radius, tip.lineage_depth):
            tip.alive = False
            return []

        seg = Segment(start=start, end=end, radius=tip.radius, parent_id=tip_index)
        self.segments.append(seg)

        tip.position = end
        tip.direction = d
        tip.radius = new_radius
        tip.age += 1

        spawned: List[Tip] = []

        can_branch = (
            tip.lineage_depth < self.params.max_depth
            and len([t for t in self.tips if t.alive]) + len(spawned) < self.params.max_tips
        )

        if can_branch and self.rng.random() < self.params.branching_probability:
            axis = orthogonal_vector(d)
            angle = math.radians(self.params.branch_angle_deg)

            if self.params.mode == "coral":
                angle *= 1.15
            elif self.params.mode == "spiral":
                angle *= 0.85

            dir1 = normalize(rotate_vector(d, axis, angle))
            child1 = Tip(
                position=end.copy(),
                direction=dir1,
                radius=max(new_radius * 0.85, self.params.min_radius),
                age=0,
                alive=True,
                lineage_depth=tip.lineage_depth + 1,
            )
            spawned.append(child1)

            if self.rng.random() < self.params.split_probability:
                dir2 = normalize(rotate_vector(d, axis, -angle))
                child2 = Tip(
                    position=end.copy(),
                    direction=dir2,
                    radius=max(new_radius * 0.82, self.params.min_radius),
                    age=0,
                    alive=True,
                    lineage_depth=tip.lineage_depth + 1,
                )
                spawned.append(child2)

        return spawned

    def step(self) -> None:
        alive_tips = [t for t in self.tips if t.alive]
        if not alive_tips:
            return

        new_tips: List[Tip] = []

        # Use a snapshot so newly spawned tips grow next step, not immediately.
        for idx, tip in list(enumerate(self.tips)):
            if not tip.alive:
                continue
            spawned = self._grow_tip_once(tip, idx)
            new_tips.extend(spawned)

        self.tips.extend(new_tips)

    def run(self) -> None:
        for _ in range(self.params.steps):
            if not any(t.alive for t in self.tips):
                break
            self.step()

    def build_tube_mesh(self, sides: int = 10) -> pv.PolyData:
        """
        Convert segments to a combined tube mesh.
        """
        if not self.segments:
            return pv.PolyData()

        meshes = []

        for seg in self.segments:
            pts = np.vstack([seg.start, seg.end])
            line = pv.lines_from_points(pts)
            tube = line.tube(radius=max(seg.radius, self.params.min_radius), n_sides=sides)
            meshes.append(tube)

        merged = meshes[0]
        for mesh in meshes[1:]:
            merged = merged.merge(mesh)

        return merged

    def build_tip_points(self) -> pv.PolyData:
        alive_positions = [t.position for t in self.tips if t.alive]
        if not alive_positions:
            return pv.PolyData()

        pts = np.array(alive_positions)
        poly = pv.PolyData(pts)
        return poly

def visualize_engine(engine: MorphogenesisEngine, title: str = "Morphogenesis") -> None:
    mesh = engine.build_tube_mesh()
    tips = engine.build_tip_points()

    plotter = pv.Plotter(window_size=(1200, 900))
    plotter.add_title(title, font_size=14)

    if mesh.n_points > 0:
        plotter.add_mesh(mesh, smooth_shading=True)

    if tips.n_points > 0:
        plotter.add_mesh(tips, render_points_as_spheres=True, point_size=10)

    plotter.add_axes()
    plotter.show_grid()
    plotter.show()


def animate_growth(params: GrowthParams, every_n_steps: int = 4) -> None:
    engine = MorphogenesisEngine(params)

    plotter = pv.Plotter(window_size=(1200, 900))
    plotter.add_title(f"Animated Growth: {params.mode}", font_size=14)
    plotter.add_axes()
    plotter.show_grid()

    mesh_actor = None
    tip_actor = None

    plotter.show(auto_close=False, interactive_update=True)

    for i in range(params.steps):
        if not any(t.alive for t in engine.tips):
            break

        engine.step()

        if i % every_n_steps == 0 or i == params.steps - 1:
            mesh = engine.build_tube_mesh()
            tips = engine.build_tip_points()

            if mesh_actor is not None:
                plotter.remove_actor(mesh_actor)
            if tip_actor is not None:
                plotter.remove_actor(tip_actor)

            if mesh.n_points > 0:
                mesh_actor = plotter.add_mesh(mesh, smooth_shading=True)
            if tips.n_points > 0:
                tip_actor = plotter.add_mesh(
                    tips,
                    render_points_as_spheres=True,
                    point_size=10,
                )

            plotter.render()

    plotter.close()

Meaning

branching_probability: how often branches split

step_size: growth amount per step

upward_bias: tendency to grow upward

radial_bias: tendency to spread away from center

twist_strength: tendency to spiral

noise_strength: randomness

branch_angle_deg: branch divergence angle

def prompt_to_params(prompt: str) -> GrowthParams:
    prompt = prompt.lower()

    if "coral" in prompt:
        params = coral_preset()
    elif "spiral" in prompt:
        params = spiral_preset()
    else:
        params = tree_preset()

    if "dense" in prompt:
        params.branching_probability *= 1.4
    if "tall" in prompt:
        params.upward_bias *= 1.2
        params.steps = int(params.steps * 1.3)
    if "wild" in prompt:
        params.noise_strength *= 1.5

    return params

import numpy as np
import random
from presets import GrowthParams


class MorphogenesisEngine:

    def __init__(self,params:GrowthParams):
        self.params=params
        self.points=[np.array([0,0,0])]
        self.directions=[np.array([0,0,1])]
        self.edges=[]


    def random_unit(self):
        v=np.random.normal(size=3)
        return v/np.linalg.norm(v)


    def step(self):

        new_points=[]
        new_dirs=[]

        for p,d in zip(self.points,self.directions):

            d=d+self.random_unit()*self.params.noise_strength
            d=d/np.linalg.norm(d)

            new_p=p+d*self.params.step_size

            new_points.append(new_p)
            new_dirs.append(d)

            self.edges.append((p,new_p))

            if random.random()<self.params.branching_probability:

                new_dirs.append(self.random_unit())
                new_points.append(new_p)

        self.points=new_points
        self.directions=new_dirs


    def run(self):

        for i in range(self.params.steps):
            self.step()

        return self.edges

from __future__ import annotations

from dataclasses import dataclass, field
from typing import Dict
import math


def clamp(x: float, lo: float = 0.0, hi: float = 1.0) -> float:
    return max(lo, min(hi, x))


def sigmoid(x: float) -> float:
    return 1.0 / (1.0 + math.exp(-x))


@dataclass
class GRNState:
    """
    Minimal gene-regulatory state for one growing tip.
    Values are normalized to roughly [0, 1].
    """
    growth: float = 0.6
    branch: float = 0.4
    adhesion: float = 0.5
    polarity: float = 0.7
    differentiation: float = 0.2

    def as_dict(self) -> Dict[str, float]:
        return {
            "growth": self.growth,
            "branch": self.branch,
            "adhesion": self.adhesion,
            "polarity": self.polarity,
            "differentiation": self.differentiation,
        }


@dataclass
class SignalInput:
    """
    External environment sensed by a tip.
    """
    nutrient: float = 0.5
    morphogen_a: float = 0.5
    morphogen_b: float = 0.5
    crowding: float = 0.0
    height: float = 0.0


def update_grn(state: GRNState, signal: SignalInput, dt: float = 1.0) -> GRNState:
    """
    A simple hand-designed GRN update.
    You can later replace this with:
    - ODEs
    - Boolean / probabilistic logic
    - learned models from single-cell or spatial data
    """

    g = state.growth
    b = state.branch
    a = state.adhesion
    p = state.polarity
    d = state.differentiation

    # External influences:
    # nutrient tends to increase growth
    # morphogen_a tends to promote branching
    # morphogen_b tends to promote polarity and differentiation
    # crowding suppresses growth and branching
    # height slightly increases differentiation

    dg = (
        1.4 * signal.nutrient
        - 0.8 * signal.crowding
        + 0.2 * b
        - 0.4 * d
        - 0.2 * g
    )

    db = (
        1.3 * signal.morphogen_a
        - 0.9 * signal.crowding
        + 0.2 * g
        - 0.3 * a
        - 0.2 * b
    )

    da = (
        0.7 * signal.crowding
        + 0.2 * signal.morphogen_b
        - 0.2 * b
        - 0.1 * a
    )

    dp = (
        1.0 * signal.morphogen_b
        + 0.3 * signal.height
        - 0.3 * signal.crowding
        - 0.15 * p
    )

    dd = (
        0.5 * signal.morphogen_b
        + 0.4 * signal.height
        - 0.2 * signal.nutrient
        + 0.2 * p
        - 0.2 * d
    )

    # Integrate with a small step
    new_state = GRNState(
        growth=clamp(g + 0.15 * dg * dt),
        branch=clamp(b + 0.15 * db * dt),
        adhesion=clamp(a + 0.15 * da * dt),
        polarity=clamp(p + 0.15 * dp * dt),
        differentiation=clamp(d + 0.15 * dd * dt),
    )
    return new_state


def grn_to_behavior(state: GRNState) -> Dict[str, float]:
    """
    Map regulatory state -> physical / developmental behavior.
    """

    branching_probability = clamp(0.02 + 0.35 * state.branch + 0.10 * state.growth)
    elongation_rate = 0.08 + 0.35 * state.growth
    upward_bias = clamp(0.15 + 0.80 * state.polarity)
    thickness_decay = clamp(0.003 + 0.02 * state.differentiation, 0.003, 0.06)
    noise_strength = clamp(0.35 - 0.20 * state.polarity + 0.10 * state.branch, 0.02, 0.45)
    split_angle_deg = 12.0 + 42.0 * state.branch
    persistence = clamp(0.25 + 0.65 * state.polarity)

    return {
        "branching_probability": branching_probability,
        "elongation_rate": elongation_rate,
        "upward_bias": upward_bias,
        "thickness_decay": thickness_decay,
        "noise_strength": noise_strength,
        "split_angle_deg": split_angle_deg,
        "persistence": persistence,
    }

from __future__ import annotations

from dataclasses import dataclass
import numpy as np
from grn import SignalInput


def clamp(x: float, lo: float = 0.0, hi: float = 1.0) -> float:
    return max(lo, min(hi, x))


@dataclass
class SignalFieldParams:
    nutrient_center: np.ndarray
    morphogen_a_center: np.ndarray
    morphogen_b_axis: np.ndarray
    nutrient_decay: float = 0.12
    morphogen_a_decay: float = 0.10
    base_height_scale: float = 0.04


class SignalField:
    """
    Provides external signals at a point in space.
    """

    def __init__(self, params: SignalFieldParams):
        self.params = params
        self.morphogen_b_axis = self._normalize(params.morphogen_b_axis)

    @staticmethod
    def _normalize(v: np.ndarray) -> np.ndarray:
        n = np.linalg.norm(v)
        if n < 1e-12:
            return v.copy()
        return v / n

    def sample(self, position: np.ndarray, crowding: float) -> SignalInput:
        p = position

        dn = np.linalg.norm(p - self.params.nutrient_center)
        nutrient = np.exp(-self.params.nutrient_decay * dn * dn)

        da = np.linalg.norm(p - self.params.morphogen_a_center)
        morphogen_a = np.exp(-self.params.morphogen_a_decay * da * da)

        # Morphogen B is a directional / axial gradient
        axial = float(np.dot(p, self.morphogen_b_axis))
        morphogen_b = clamp(0.5 + 0.15 * axial)

        height = clamp(self.params.base_height_scale * p[2])

        return SignalInput(
            nutrient=clamp(float(nutrient)),
            morphogen_a=clamp(float(morphogen_a)),
            morphogen_b=clamp(float(morphogen_b)),
            crowding=clamp(crowding),
            height=height,
        )

from __future__ import annotations

from dataclasses import dataclass
from typing import List, Optional
import math
import random

import numpy as np

from grn import GRNState, update_grn, grn_to_behavior
from signals import SignalField


def normalize(v: np.ndarray) -> np.ndarray:
    n = np.linalg.norm(v)
    if n < 1e-12:
        return v.copy()
    return v / n


def random_unit_vector() -> np.ndarray:
    v = np.random.normal(size=3)
    return normalize(v)


def orthogonal_vector(v: np.ndarray) -> np.ndarray:
    v = normalize(v)
    if abs(v[0]) < 0.9:
        other = np.array([1.0, 0.0, 0.0])
    else:
        other = np.array([0.0, 1.0, 0.0])
    return normalize(np.cross(v, other))


def rotate_vector(vec: np.ndarray, axis: np.ndarray, angle_rad: float) -> np.ndarray:
    axis = normalize(axis)
    cos_a = math.cos(angle_rad)
    sin_a = math.sin(angle_rad)
    return (
        vec * cos_a
        + np.cross(axis, vec) * sin_a
        + axis * np.dot(axis, vec) * (1.0 - cos_a)
    )


def distance_point_to_segment(p: np.ndarray, a: np.ndarray, b: np.ndarray) -> float:
    ab = b - a
    denom = np.dot(ab, ab)
    if denom < 1e-12:
        return float(np.linalg.norm(p - a))
    t = np.dot(p - a, ab) / denom
    t = max(0.0, min(1.0, t))
    closest = a + t * ab
    return float(np.linalg.norm(p - closest))


@dataclass
class Segment:
    start: np.ndarray
    end: np.ndarray
    radius: float
    depth: int


@dataclass
class Tip:
    position: np.ndarray
    direction: np.ndarray
    radius: float
    grn: GRNState
    age: int = 0
    alive: bool = True
    depth: int = 0


@dataclass
class EngineParams:
    steps: int = 300
    dt: float = 1.0
    max_tips: int = 120
    max_depth: int = 8
    initial_radius: float = 0.08
    min_radius: float = 0.01
    collision_distance: float = 0.10
    collision_strength: float = 0.6
    seed: int = 42


class GRNMorphogenesisEngine:
    def __init__(self, params: EngineParams, field: SignalField):
        self.params = params
        self.field = field
        self.rng = random.Random(params.seed)
        np.random.seed(params.seed)

        self.origin = np.array([0.0, 0.0, 0.0], dtype=float)
        self.tips: List[Tip] = []
        self.segments: List[Segment] = []

        self._initialize()

    def _initialize(self) -> None:
        root = Tip(
            position=self.origin.copy(),
            direction=np.array([0.0, 0.0, 1.0], dtype=float),
            radius=self.params.initial_radius,
            grn=GRNState(
                growth=0.65,
                branch=0.30,
                adhesion=0.40,
                polarity=0.75,
                differentiation=0.10,
            ),
            age=0,
            alive=True,
            depth=0,
        )
        self.tips.append(root)

    def _estimate_crowding(self, position: np.ndarray) -> float:
        if not self.segments:
            return 0.0

        score = 0.0
        for seg in self.segments[-300:]:
            dist = distance_point_to_segment(position, seg.start, seg.end)
            if dist < self.params.collision_distance:
                score += (self.params.collision_distance - dist) / self.params.collision_distance

        return max(0.0, min(1.0, score))

    def _collision_repel(self, position: np.ndarray, proposed_dir: np.ndarray) -> np.ndarray:
        repel = np.zeros(3, dtype=float)
        test_point = position + proposed_dir * 0.15

        for seg in self.segments[-300:]:
            dist = distance_point_to_segment(test_point, seg.start, seg.end)
            if dist < self.params.collision_distance:
                midpoint = 0.5 * (seg.start + seg.end)
                away = test_point - midpoint
                if np.linalg.norm(away) > 1e-12:
                    strength = (self.params.collision_distance - dist) / self.params.collision_distance
                    repel += normalize(away) * strength

        if np.linalg.norm(repel) > 1e-12:
            proposed_dir = normalize(proposed_dir + self.params.collision_strength * repel)

        return proposed_dir

    def _compute_growth_direction(self, tip: Tip, behavior: dict) -> np.ndarray:
        direction = tip.direction.copy()

        # Upward polarity term
        direction += np.array([0.0, 0.0, 1.0]) * behavior["upward_bias"]

        # Nutrient attractor
        nutrient_center = self.field.params.nutrient_center
        toward_nutrient = nutrient_center - tip.position
        if np.linalg.norm(toward_nutrient) > 1e-12:
            direction += 0.25 * normalize(toward_nutrient)

        # Morphogen A outward branching region
        outward = tip.position.copy()
        outward[2] = 0.0
        if np.linalg.norm(outward) > 1e-12:
            direction += 0.10 * normalize(outward)

        # Persistence + noise
        persistence = behavior["persistence"]
        noise_strength = behavior["noise_strength"]
        direction = normalize(persistence * direction + (1.0 - persistence) * random_unit_vector())
        direction = normalize(direction + noise_strength * random_unit_vector())

        # Collision avoidance
        direction = self._collision_repel(tip.position, direction)

        return normalize(direction)

    def _spawn_branch(self, tip: Tip, new_pos: np.ndarray, parent_dir: np.ndarray, angle_deg: float) -> List[Tip]:
        if tip.depth >= self.params.max_depth:
            return []

        axis = orthogonal_vector(parent_dir)
        angle = math.radians(angle_deg)

        children: List[Tip] = []

        dir1 = normalize(rotate_vector(parent_dir, axis, angle))
        child1 = Tip(
            position=new_pos.copy(),
            direction=dir1,
            radius=max(tip.radius * 0.88, self.params.min_radius),
            grn=GRNState(**tip.grn.as_dict()),
            age=0,
            alive=True,
            depth=tip.depth + 1,
        )
        children.append(child1)

        if self.rng.random() < 0.45:
            dir2 = normalize(rotate_vector(parent_dir, axis, -angle))
            child2 = Tip(
                position=new_pos.copy(),
                direction=dir2,
                radius=max(tip.radius * 0.84, self.params.min_radius),
                grn=GRNState(**tip.grn.as_dict()),
                age=0,
                alive=True,
                depth=tip.depth + 1,
            )
            children.append(child2)

        return children

    def step(self) -> None:
        if not any(t.alive for t in self.tips):
            return

        spawned: List[Tip] = []

        for tip in list(self.tips):
            if not tip.alive:
                continue

            crowding = self._estimate_crowding(tip.position)
            signal = self.field.sample(tip.position, crowding)
            tip.grn = update_grn(tip.grn, signal, dt=self.params.dt)
            behavior = grn_to_behavior(tip.grn)

            step_len = behavior["elongation_rate"]
            new_dir = self._compute_growth_direction(tip, behavior)
            new_pos = tip.position + step_len * new_dir

            # Add segment
            self.segments.append(
                Segment(
                    start=tip.position.copy(),
                    end=new_pos.copy(),
                    radius=tip.radius,
                    depth=tip.depth,
                )
            )

            # Update tip
            tip.position = new_pos
            tip.direction = new_dir
            tip.radius = max(self.params.min_radius, tip.radius - behavior["thickness_decay"])
            tip.age += 1

            # Kill conditions
            if tip.radius <= self.params.min_radius + 1e-9:
                tip.alive = False
                continue

            if tip.depth > self.params.max_depth:
                tip.alive = False
                continue

            # Branching
            alive_count = sum(1 for t in self.tips if t.alive) + len(spawned)
            if alive_count < self.params.max_tips:
                if self.rng.random() < behavior["branching_probability"]:
                    new_children = self._spawn_branch(
                        tip=tip,
                        new_pos=new_pos,
                        parent_dir=new_dir,
                        angle_deg=behavior["split_angle_deg"],
                    )
                    spawned.extend(new_children)

        self.tips.extend(spawned)

    def run(self) -> None:
        for _ in range(self.params.steps):
            if not any(t.alive for t in self.tips):
                break
            self.step()

    def get_segment_arrays(self) -> List[dict]:
        result = []
        for s in self.segments:
            result.append(
                {
                    "x1": float(s.start[0]),
                    "y1": float(s.start[1]),
                    "z1": float(s.start[2]),
                    "x2": float(s.end[0]),
                    "y2": float(s.end[1]),
                    "z2": float(s.end[2]),
                    "radius": float(s.radius),
                    "depth": int(s.depth),
                }
            )
        return result

from __future__ import annotations

import numpy as np
import pyvista as pv
from engine import GRNMorphogenesisEngine


def build_tube_mesh(engine: GRNMorphogenesisEngine, n_sides: int = 10) -> pv.PolyData:
    if not engine.segments:
        return pv.PolyData()

    meshes = []

    for seg in engine.segments:
        pts = np.vstack([seg.start, seg.end])
        line = pv.lines_from_points(pts)
        tube = line.tube(radius=max(seg.radius, 0.01), n_sides=n_sides)
        meshes.append(tube)

    merged = meshes[0]
    for mesh in meshes[1:]:
        merged = merged.merge(mesh)

    return merged


def show_engine(engine: GRNMorphogenesisEngine, title: str = "GRN Morphogenesis") -> None:
    mesh = build_tube_mesh(engine)

    plotter = pv.Plotter(window_size=(1200, 900))
    plotter.add_title(title, font_size=14)

    if mesh.n_points > 0:
        plotter.add_mesh(mesh, smooth_shading=True)

    plotter.add_axes()
    plotter.show_grid()
    plotter.show()
